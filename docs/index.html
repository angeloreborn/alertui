Simple Popup 
ui.popup({
    body: "Hello, there!",
    actions: [{
        title: "Close",
        className: "btn btn-default m-2"
    }]
})
Yes/No Popup
ui.popup({
    body: "Are you sure you want to do that?",
    actions: [
    {
        title: "no",
        className: "btn btn-default m-2"
    },
    {
        title: "yes",
        className: "btn btn-primary m-2",
        onClick: doSomething
    }
]
})

function doSomething(){
    console.log("I did something");
}
Multiselect Popup
ui.popup({
    body: "Whats your favourite mode of transport?",
    actions: [
        {
            title: "Car",
            className: "btn btn-pimary m-2",
            onClick: ()=>handleTransportMode("Car")
        },
        {
            title: "Bicycle",
            className: "btn btn-pimary m-2",
            onClick: ()=>handleTransportMode("Bicycle")
        },
        {
            title: "Walking",
            className: "btn btn-pimary m-2",
            onClick: ()=>handleTransportMode("Walking")
        },
    ]
})

function handleTransportMode(transportMode) {
    ui.popup({
        body: `You selected: <b>${transportMode}</b>`,
        actions:[{
            title:"close",
            className: "btn btn-default"
        }]
    })
}
